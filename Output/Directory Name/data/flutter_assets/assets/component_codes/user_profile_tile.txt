import 'package:flutter/material.dart';

/// A user profile tile with avatar, name, and optional subtitle.
class UserProfileTile extends StatelessWidget {
  /// Create a user profile tile.
  ///
  /// Parameters:
  ///   - avatar: The user's avatar widget.
  ///   - name: The user's name.
  ///   - subtitle: Optional subtitle text (e.g., email or role).
  ///   - onTap: Callback function when the tile is tapped.
  ///   - trailing: Optional trailing widget.
  const UserProfileTile({
    super.key,
    required this.avatar,
    required this.name,
    this.subtitle,
    this.onTap,
    this.trailing,
  });

  final Widget avatar;
  final String name;
  final String? subtitle;
  final VoidCallback? onTap;
  final Widget? trailing;

  @override
  Widget build(BuildContext context) {
    return ListTile(
      leading: avatar,
      title: Text(
        name,
        style: const TextStyle(
          fontWeight: FontWeight.bold,
        ),
      ),
      subtitle: subtitle != null ? Text(subtitle!) : null,
      trailing: trailing,
      onTap: onTap,
    );
  }
}
